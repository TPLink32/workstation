内容简介 

 #include <sys/types.h>  
 #include <sys/ipc.h>  
 #include <sys/msg.h>

int msgctl(int  msqid ,   int  cmd ,   struct msqid_ds * buf );

struct ipc_perm {
    key_t key;            /* Key supplied to msgget() */
